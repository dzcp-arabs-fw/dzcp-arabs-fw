Applying Effects

Use the effect system like this:

using DZCP.Effects;

public class SmokeEffectDemo
{
    public void TriggerSmoke()
    {
        GameEffects.PlaySmokeEffect(10.5f, 3.2f, -5.0f, 5.0f);
    }
}
