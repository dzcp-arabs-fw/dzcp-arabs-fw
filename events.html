Handling Events

Subscribe to events in DZCP like this:

using DZCP.Platform;

namespace DZCP.Events
{
    public static class EventHandlers
    {
        public static void RegisterEvents()
        {
            GameEvents.OnPlayerJoin += PlayerJoined;
        }

        private static void PlayerJoined(string playerName)
        {
            Console.WriteLine($"[DZCP] Player {playerName} has joined the server!");
        }
    }
}
